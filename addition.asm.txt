section .data
    res db 'Result =', 0
    reslen equ $ - res

    result db 0

section .text
    global _start

_start:
    ;Passing values to registers
    mov al, 5       
    mov bl, 4     
    ;Performing the operation
    add al, bl  
    ;converting to ASCII, other wise gibberish will be outputted
    add al, '0' 
    ;assigning result the answer
    mov [result], al
    ;printing the statement
    mov eax, 4       
    mov ebx, 1      
    mov ecx, res
    mov edx, reslen
    int 0x80
    ;printing the result
    mov eax, 4
    mov ebx, 1
    mov ecx, result
    mov edx, 1
    int 0x80
    ;termination
    mov eax, 1      
    mov ebx, 0
    int 0x80
